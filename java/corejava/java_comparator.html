Comparator

  • Comparator is in java.util.package.
  • Comparator provides two methods compare() and equals() methods.
  • It provides multiple sorting sequence.

package com.sort;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.List;
public class PriceComparator {
public static void main(String[] args) {
	naturalOrderingNumbers();
	naturalOrderingStrings();
	comparatorSortIntegers();
    comparatorSortStrings();
}
// Sorting the Strings using comparable, Natural ordering
private static void naturalOrderingNumbers() {
	List intList = new ArrayList();
	intList.add(10);
	intList.add(23);
	intList.add(11);
	intList.add(19);
	intList.add(23);
	Collections.sort(intList);
	System.out.println("Sort in order of Price -Natural Ordering ");
	for (Integer i : intList) {
		System.out.println(i);
	}
	System.out.println("\n");
}
// Sorting the numbers using comparable, Natural ordering
private static void naturalOrderingStrings() {
	List strList = new ArrayList();
	strList.add("Audi");
	strList.add("BMW");
	strList.add("Hyndai");
	strList.add("Suzuki");
	strList.add("BMW");
	strList.add("Ford");
	System.out.println("Sort in order of Make -Natural Ordering ");
	Collections.sort(strList);
	for (String str : strList) {
		System.out.println(str);
	}
	System.out.println("\n");
}
// Sorting in order of price using Comparator and anonymous inner class
private static void comparatorSortIntegers() {
	List plist = new ArrayList();
	plist.add(new Price(10, "Audi"));
	plist.add(new Price(20, "BMW"));
	plist.add(new Price(36, "Hyundai"));
	plist.add(new Price(23, "Audi"));
	plist.add(new Price(27, "Benz"));
	Collections.sort(plist, new Comparator() {
		@Override
		public int compare(Price p1, Price p2) {
			if (p1 == null || p2 == null) {
			throw new NullPointerException("Objects are null");
			} else if (!p1.getClass().equals(p2.getClass())) {
				throw new ClassCastException("Objects are not of the same Type, Not mutually comparable");
			} else if (p1.getPrice() > p2.getPrice()) {
				return 1;
			} else if (p1.getPrice() < p2.getPrice()) {
				return -1;
			} else
				return 0;
		}});
	System.out.println("Sort in order of Price -Comparator and anonymous Inner class ");
	for (Price i : plist) {
			System.out.println(i.getPrice());
	}
	System.out.println("\n");
}
	// Sorting in order of Make using Comparator and anonymous inner class
private static void comparatorSortStrings() {
	List plist = new ArrayList();
	Price p1 = new Price(10, "Audi");
	plist.add(new Price(20, "BMW"));
	plist.add(new Price(36, "Hyundai"));
	plist.add(new Price(23, "Audi"));
	plist.add(new Price(27, "Benz"));
	plist.add(p1);
	Collections.sort(plist, new Comparator() {
	@Override
	public int compare(Price p1, Price p2) {
		if (p1 == null || p2 == null)
			throw new NullPointerException("Object is null");
		else if (!p1.getClass().equals(p2.getClass()))
			throw new ClassCastException("Objects are not mutually comparable");
		else
			return (p1.getBrand().compareTo(p2.getBrand()));
		}});
	System.out.println("Sort in order of Make -Comparator and anonymous Inner class ");
	for (Price p : plist)
		System.out.println(p.getBrand());
	System.out.println("\n");
	}
}
